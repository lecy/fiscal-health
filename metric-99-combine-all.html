rescale <- function(x)
{
   x[ is.na(x) ] <- median(x, na.rm=T )

   x.05 <- quantile( x, 0.05 )
   x.95 <- quantile( x, 0.95 )

   x[ x < x.05 ] <- x.05
   x[ x > x.95 ] <- x.95 
   
   x <- scales::rescale( x, to=c(1,100) )
   
   if( abs( mean(x) - median(x) ) > 5 )
   { x <- log(x) }

   x <- scale(x)

   return(x)
}
